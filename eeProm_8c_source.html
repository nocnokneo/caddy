106 uint8_t EEPROM_read(unsigned int uiAddress)
107 {
108  // Wait for completion of previous write
109  while (EECR & (1 << EEWE)) ;
110  // Set up address register
111  EEAR = uiAddress;
112  // Start eeprom read by writing EERE
113  EECR |= (1 << EERE);
114  // Return data from data register
115  return EEDR;
116 }

118 void EEPROM_write(unsigned int uiAddress, uint8_t ucData)
119 {
120  // Wait for completion of previous write
121  while (EECR & (1 << EEWE)) ;
122  // Set up address and data registers
123  EEAR = uiAddress;
124  EEDR = ucData;
125  // Write logical one to EEMWE
126  EECR |= (1 << EEMWE);
127  // Start eeprom write by setting EEWE
128  EECR |= (1 << EEWE);
129  // EEMWE and EEWE are automatically cleared back to 0 by hardware
130 }
